376 dst[1][0] = src[0][1];

386 dst[1][2] * dst[1][2]),

395 dst[1][1] = 1.0f / (dst[1][1] * lengths[1]);
396 dst[1][2] = 1.0f / (dst[1][2] * lengths[1]);
