344 biggestIndex = 1;

349 biggestIndex = 2;

359
360 switch (biggestIndex)
361 {
362 case 0:
363 return qua<T>(biggestVal, (m[1][2] - m[2][1]) * mult, (m[2][0] - m[0][2]) * mult, (m[0][1] - m[1][0]) * mult);
364 case 1:
